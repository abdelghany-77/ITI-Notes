2.4 Keys in Database Design

Key Type Definition Properties
Primary Key (PK) Uniquely identifies each row in a table â€¢ Must be UNIQUE
â€¢ Cannot be NULL
â€¢ Only one per table
â€¢ Usually an ID column
Foreign Key (FK) Links two tables together â€¢ References PK of another table

â€¢ Can be NULL

Referential Integrity ensures that relationships between tables remain consistent. A foreign key value must either match a primary key value or be NULL.

4.5 Transaction Management

Transactions ensure data integrity by grouping multiple DML operations into a single atomic unit. Either all operations succeed (COMMIT) or all fail (ROLLBACK). 

-- Start transaction
mysql> START TRANSACTION;
-- Or: BEGIN;

-- Perform operations
mysql> UPDATE accounts SET balance = balance - 1000 WHERE account_id = 1;
mysql> UPDATE accounts SET balance = balance + 1000 WHERE account_id = 2;

-- Commit if all successful
mysql> COMMIT;

-- Or rollback if something went wrong
mysql> ROLLBACK;

-- Set savepoint (partial rollback)
mysql> START TRANSACTION;
mysql> INSERT INTO orders (customer_id, total_amount) VALUES (1, 100);
mysql> SAVEPOINT sp1;
mysql> INSERT INTO orders (customer_id, total_amount) VALUES (2, 200);
mysql> SAVEPOINT sp2;
mysql> INSERT INTO orders (customer_id, total_amount) VALUES (3, 300);

-- Rollback to savepoint (keeps first two inserts)
mysql> ROLLBACK TO SAVEPOINT sp2;

mysql> COMMIT;

-- Auto-commit mode
mysql> SET autocommit = 0;  -- Disable auto-commit
-- Now must manually COMMIT or ROLLBACK

mysql> SET autocommit = 1;  -- Enable auto-commit (default)

-- Transaction with error handling
DELIMITER $$
CREATE PROCEDURE transfer_funds(
    IN from_account INT,
    IN to_account INT,
    IN amount DECIMAL(10,2)
)
BEGIN
    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        ROLLBACK;
        SELECT 'Transaction failed and rolled back' AS message;
    END;
    
    START TRANSACTION;
    
    UPDATE accounts 
    SET balance = balance - amount 
    WHERE account_id = from_account AND balance >= amount;
    
    IF ROW_COUNT() = 0 THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Insufficient funds';
    END IF;
    
    UPDATE accounts 
    SET balance = balance + amount 
    WHERE account_id = to_account;
    
    COMMIT;
    SELECT 'Transaction successful' AS message;
END$$
DELIMITER ;

Order Clause Purpose

A Trigger is a stored procedure that automatically executes when a specific event occurs on a table (INSERT, UPDATE, or DELETE). Triggers are useful for maintaining audit logs, enforcing business rules, and data validation.

CREATE TRIGGER before_employee_insert
BEFORE INSERT ON employees
FOR EACH ROW
BEGIN
    -- Automatically set email to lowercase
    SET NEW.email = LOWER(NEW.email);
    
    -- Set created_at timestamp
    SET NEW.created_at = NOW();
END$$
DELIMITER ;

-- AFTER INSERT trigger (audit log)

AFTER INSERT ON employees
FOR EACH ROW
BEGIN
    INSERT INTO employee_audit_log (
        employee_id,
        action,
        action_date,
        action_by
    )
    VALUES (
        NEW.employee_id,
        'INSERT',
        NOW(),
        USER()
    );
END$$
DELIMITER ;

-- BEFORE UPDATE trigger
DELIMITER $$
CREATE TRIGGER before_employee_update
BEFORE UPDATE ON employees
FOR EACH ROW
BEGIN
    -- Prevent salary decrease

        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Salary cannot be decreased';

            OLD.salary,
            NEW.salary,

            USER()

9.4 Window Functions

Window Functions (MySQL 8.0+) perform calculations across rows related to the current row, without collapsing rows like GROUP BY does. They're powerful for analytics, rankings, and running totals.

Window Function Syntax

function_name() OVER (
    [PARTITION BY column1, column2, ...]
    [ORDER BY column1 [ASC|DESC], ...]
    [ROWS/RANGE frame_specification]
)

Components:
- PARTITION BY: Divides rows into groups (like GROUP BY but keeps detail)
- ORDER BY: Defines order within partition
- Frame: Defines subset of partition for calculation

-- ROW_NUMBER() - Unique sequential number
mysql> SELECT 
    employee_id,
    first_name,
    last_name,
    salary,
    ROW_NUMBER() OVER (ORDER BY salary DESC) AS salary_rank
FROM employees;

-- RANK() - Rank with gaps for ties
mysql> SELECT 
    employee_id,
    first_name,
    salary,
    RANK() OVER (ORDER BY salary DESC) AS rank_with_gaps
FROM employees;
-- If two employees have same salary (rank 1), next rank is 3

-- DENSE_RANK() - Rank without gaps
mysql> SELECT 
    employee_id,
    first_name,
    salary,
    DENSE_RANK() OVER (ORDER BY salary DESC) AS rank_no_gaps
FROM employees;
-- If two employees have same salary (rank 1), next rank is 2

-- Partition by department
mysql> SELECT 
    department_id,
    employee_id,
    first_name,
    salary,
    RANK() OVER (PARTITION BY department_id ORDER BY salary DESC) AS dept_rank
FROM employees;
-- Ranks reset for each department

-- NTILE() - Divide into N groups
mysql> SELECT 
    employee_id,
    first_name,
    salary,
    NTILE(4) OVER (ORDER BY salary) AS salary_quartile
FROM employees;
-- Divides employees into 4 equal groups

-- LAG() - Access previous row
mysql> SELECT 
    order_id,
    order_date,
    total_amount,
    LAG(total_amount) OVER (ORDER BY order_date) AS previous_order_amount,
    total_amount - LAG(total_amount) OVER (ORDER BY order_date) AS difference
FROM orders;

-- LEAD() - Access next row
mysql> SELECT 
    order_id,
    order_date,
    total_amount,
    LEAD(total_amount) OVER (ORDER BY order_date) AS next_order_amount
FROM orders;

-- FIRST_VALUE() and LAST_VALUE()
mysql> SELECT 
    employee_id,
    department_id,
    salary,
    FIRST_VALUE(salary) OVER (PARTITION BY department_id ORDER BY salary) AS lowest_in_dept,
    LAST_VALUE(salary) OVER (
        PARTITION BY department_id 
        ORDER BY salary
        RANGE BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING
    ) AS highest_in_dept
FROM employees;

-- Running total (SUM with frame)
mysql> SELECT 
    order_date,
    total_amount,
    SUM(total_amount) OVER (
        ORDER BY order_date
        ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
    ) AS running_total
FROM orders;

-- Moving average (last 3 rows)
mysql> SELECT 
    order_date,
    total_amount,
    AVG(total_amount) OVER (
        ORDER BY order_date
        ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
    ) AS moving_avg_3
FROM orders;

-- Aggregate functions as window functions
mysql> SELECT 
    department_id,
    employee_id,
    first_name,
    salary,
    AVG(salary) OVER (PARTITION BY department_id) AS dept_avg_salary,
    salary - AVG(salary) OVER (PARTITION BY department_id) AS diff_from_avg,
    COUNT(*) OVER (PARTITION BY department_id) AS dept_employee_count
FROM employees;

-- Complex example: Top 3 products per category
mysql> SELECT * FROM (
    SELECT 
        c.category_name,
        p.product_name,
        p.price,
        ROW_NUMBER() OVER (PARTITION BY c.category_id ORDER BY p.price DESC) AS price_rank
    FROM products p
    JOIN categories c ON p.category_id = c.category_id
) ranked
WHERE price_rank <= 3;

-- Year-over-year comparison
mysql> SELECT 
    YEAR(order_date) AS year,
    MONTH(order_date) AS month,
    SUM(total_amount) AS monthly_revenue,
    LAG(SUM(total_amount), 12) OVER (ORDER BY YEAR(order_date), MONTH(order_date)) AS last_year_revenue,
    SUM(total_amount) - LAG(SUM(total_amount), 12) OVER (ORDER BY YEAR(order_date), MONTH(order_date)) AS yoy_growth
FROM orders
GROUP BY YEAR(order_date), MONTH(order_date);

    product_name,

    SUM(CASE WHEN MONTH(order_date) = 2 THEN total ELSE 0 END) AS feb,

-- Find gaps in sequential IDs

     FROM orders t3

    SELECT 1 FROM orders t2

)
AND t1.order_id < (SELECT MAX(order_id) FROM orders);

WITH numbered AS (
    SELECT 
        customer_id,
        login_date,
        ROW_NUMBER() OVER (PARTITION BY customer_id ORDER BY login_date) AS rn,

            ROW_NUMBER() OVER (PARTITION BY customer_id ORDER BY login_date) DAY
        ) AS group_date
    FROM customer_logins

    customer_id,
    MIN(login_date) AS streak_start,
    MAX(login_date) AS streak_end,

GROUP BY customer_id, group_date
HAVING COUNT(*) >= 7  -- 7+ day streaks

Day 3.1: Transactions

A Transaction is a set of SQL statements that execute as a unit. Either all statements succeed (COMMIT), or none of them have any effect (ROLLBACK). Transactions ensure data integrity and consistency.

Example Scenario: Money Transfer

Moataz writes a check to Ali for $100. Moataz's account should be decremented by $100 and Ali's account incremented by the same amount. If a crash occurs between the two operations, the transaction ensures either both operations complete or neither does. 

-- Money transfer example WITHOUT transaction (UNSAFE!)
mysql> UPDATE account SET balance = balance - 100 WHERE name = 'Moataz';
mysql> UPDATE account SET balance = balance + 100 WHERE name = 'Ali';
-- If crash occurs between statements, data is inconsistent!

-- Money transfer WITH transaction (SAFE)
mysql> START TRANSACTION;
mysql> UPDATE account SET balance = balance - 100 WHERE name = 'Moataz';
mysql> UPDATE account SET balance = balance + 100 WHERE name = 'Ali';
mysql> COMMIT;
-- Both operations complete or neither does

-- Create test table
mysql> CREATE TABLE t (name CHAR(20), UNIQUE (name)) ENGINE = InnoDB;

-- Transaction with BEGIN TRANSACTION
mysql> BEGIN TRANSACTION;
mysql> INSERT INTO t SET name = 'Islam';
mysql> INSERT INTO t SET name = 'Moataz';
mysql> COMMIT;
mysql> SELECT * FROM t;

-- Transaction with ROLLBACK
mysql> START TRANSACTION;
mysql> INSERT INTO t SET name = 'Ahmed';
mysql> INSERT INTO t SET name = 'Sara';
mysql> ROLLBACK;  -- Undo both inserts
mysql> SELECT * FROM t;  -- Ahmed and Sara not inserted

-- SAVEPOINT for partial rollback
mysql> CREATE TABLE t (i INT);
mysql> BEGIN TRANSACTION;
mysql> INSERT INTO t VALUES(1);
mysql> SAVEPOINT my_savepoint;
mysql> INSERT INTO t VALUES(2);
mysql> ROLLBACK TO SAVEPOINT my_savepoint;  -- Only rollback to savepoint
mysql> INSERT INTO t VALUES(3);
mysql> COMMIT;
mysql> SELECT * FROM t;  -- Shows 1 and 3, not 2

-- Autocommit mode control
mysql> SET autocommit = 0;  -- Disable autocommit
mysql> INSERT INTO t SET name = 'Islam';
mysql> INSERT INTO t SET name = 'Moataz';
mysql> COMMIT;  -- Manually commit
mysql> SELECT * FROM t;
mysql> SET autocommit = 1;  -- Re-enable autocommit
âš ï¸ Important Transaction Notes:
  • Must use transactional storage engine (InnoDB, not MyISAM)
  • By default, MySQL runs in autocommit mode (each statement commits immediately)
  • Some statements cause implicit commit (ALTER TABLE, CREATE INDEX, DROP TABLE, etc.)
  • DDL statements wait for transactions to complete before executing

Built-in Functions modify and format result sets. They are categorized into: Single Row Functions (scalar) that work row-by-row, and Multi-Row Functions (aggregation) that work on sets of rows.

Stored Functions calculate a value to be returned to the caller for use in expressions, just like built-in functions (SIN(), COS(), etc.). Unlike stored procedures, functions must return a value and can be used directly in SELECT statements.

    DECLARE result BIGINT DEFAULT 1;

        SET counter = counter + 1;
    UNTIL counter > n
    END REPEAT;
    
    RETURN total;
END$$
DELIMITER ;

Day 3.4: Triggers (Detailed)

Triggers are stored procedures that automatically execute when a specific event (INSERT, UPDATE, DELETE) occurs on a table. The difference from stored procedures is that triggers are activated automatically by database events, not called explicitly.

Trigger Syntax Components:

  • trigger_time: BEFORE or AFTER
  • trigger_event: INSERT, UPDATE, or DELETE
  • table_name: Table the trigger is associated with
  • FOR EACH ROW: Executes once for each affected row
  • OLD: Refers to existing row before UPDATE/DELETE
  • NEW: Refers to new row after INSERT/UPDATE
-- Create audit table for tracking changes
mysql> CREATE TABLE employees_audit (
    id INT(11) NOT NULL AUTO_INCREMENT,
    employeeNumber INT(11) NOT NULL,
    lastname VARCHAR(50) NOT NULL,
    changedon DATETIME DEFAULT NULL,
    action VARCHAR(50) DEFAULT NULL,
    PRIMARY KEY (id)
);

-- BEFORE UPDATE trigger - Track changes
DELIMITER $$
CREATE TRIGGER before_employee_update
BEFORE UPDATE ON employees
FOR EACH ROW
BEGIN
    INSERT INTO employees_audit
    SET action = 'update',
        employeeNumber = OLD.employeeNumber,
        lastname = OLD.lastname,
        changedon = NOW();
END$$
DELIMITER ;

-- Test the trigger
mysql> UPDATE employees SET lastname = 'NewName' WHERE employeeNumber = 1;
mysql> SELECT * FROM employees_audit;

-- BEFORE INSERT trigger - Data validation
DELIMITER $$
CREATE TRIGGER before_employee_insert
BEFORE INSERT ON employees
FOR EACH ROW
BEGIN
    -- Ensure email is lowercase
    SET NEW.email = LOWER(NEW.email);
    
    -- Set default hire date if not provided
    IF NEW.hire_date IS NULL THEN
        SET NEW.hire_date = CURDATE();
    END IF;
    
    -- Validate salary
    IF NEW.salary < 0 THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Salary cannot be negative';
    END IF;
END$$
DELIMITER ;

-- AFTER INSERT trigger - Audit log
DELIMITER $$
CREATE TRIGGER after_employee_insert
AFTER INSERT ON employees
FOR EACH ROW

    INSERT INTO employees_audit (
        employeeNumber,
        lastname,
        action,
        changedon

        NEW.employeeNumber,
        NEW.lastname,

        NOW()
    );

Day 3.5: Backup and Restore

Database Backup is the process of creating copies of data to protect against data loss. mysqldump is the primary tool for logical backups, creating SQL files that can be used to recreate databases. 

-- Basic database backup (command line)
$ mysqldump --databases sampdb > sampdb.sql
$ mysqldump -u root -p company > company_backup.sql

-- Backup specific tables only
$ mysqldump -u root -p company employees departments > tables_backup.sql

-- Backup all databases
$ mysqldump -u root -p --all-databases > all_databases_backup.sql

-- Backup with compression (save disk space)
$ mysqldump -u root -p company | gzip > company_backup.sql.gz

-- Backup structure only (no data)
$ mysqldump -u root -p --no-data company > company_structure.sql

-- Backup data only (no structure)
$ mysqldump -u root -p --no-create-info company > company_data.sql

-- Full backup with stored procedures, triggers, events
$ mysqldump -u root -p \
  --single-transaction \    # Consistent backup without locking
  --routines \              # Include stored procedures/functions
  --triggers \              # Include triggers
  --events \                # Include scheduled events
  company > company_full_backup.sql

-- Restore database from backup
$ mysql -u root -p company < company_backup.sql
$ mysql < /tmp/sampdb.sql

-- Restore compressed backup
$ gunzip < company_backup.sql.gz | mysql -u root -p company

-- Restore all databases
$ mysql -u root -p < all_databases_backup.sql

-- Create database before restore (if doesn't exist)
mysql> CREATE DATABASE company;
$ mysql -u root -p company < company_backup.sql

Automated Backup Script (Bash):

#!/bin/bash
# Automated MySQL Backup Script

DATE=$(date +%Y%m%d_%H%M%S)
BACKUP_DIR="/backups/mysql"
DB_NAME="company"
DB_USER="backup_user"
DB_PASS="secure_password"

# Create backup directory
mkdir -p $BACKUP_DIR

# Perform backup with compression
mysqldump -u $DB_USER -p$DB_PASS \
  --single-transaction \
  --routines \
  --triggers \
  --events \
  $DB_NAME | gzip > $BACKUP_DIR/${DB_NAME}_${DATE}.sql.gz

# Delete backups older than 30 days
find $BACKUP_DIR -name "*.sql.gz" -mtime +30 -delete

echo "Backup completed: ${DB_NAME}_${DATE}.sql.gz"
