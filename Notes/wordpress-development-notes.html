Lesson 1: Introduction to WordPress

What is WordPress and why it's the most popular CMS

1.1 What is WordPress?

WordPress is a free and open-source content management system (CMS) written in PHP and paired with a MySQL database. It was created by Matt Mullenweg and Mike Little in 2003 and has since become the most popular CMS in the world, powering over 40% of all websites on the internet.

Open Source

Free to use, modify, and distribute under GPL license

User-Friendly

Intuitive admin interface for non-technical users

Extensible

Thousands of themes and plugins available

SEO-Friendly

Built-in SEO features and optimization tools

1.2 WordPress Architecture

Core Files

wp-admin, wp-content, wp-includes directories

Database

MySQL database storing posts, users, settings

Themes

Control visual appearance and layout

Plugins

Extend functionality without modifying core

Lesson 2: Installation & Setup

Setting up WordPress locally and on servers

2.1 System Requirements

PHP

Version 7.4 or higher (8.0+ recommended)

Extensions: mysqli, pdo_mysql, mbstring, curl, gd, xml, zip

MySQL

Version 5.7 or higher (8.0+ recommended)

InnoDB storage engine required

Web Server

Apache, Nginx, or IIS with mod_rewrite

URL rewriting support required

HTTPS

SSL certificate for secure connections

Let's Encrypt or commercial certificates

Minimum Hardware Requirements
  • RAM: 512MB minimum, 1GB+ recommended
  • Storage: 1GB for WordPress core, additional space for uploads/media
  • CPU: 1GHz+ processor, multiple cores recommended
  • Network: Stable internet connection for updates and plugins

2.2 Local Development Setup

Using XAMPP

XAMPP is a free and open-source cross-platform web server solution stack.

  1. Download XAMPP from apachefriends.org

  3. Download WordPress from wordpress.org
  4. Extract to htdocs folder
  5. Create database in phpMyAdmin

8.3 REST API

Creating Custom REST API Endpoints
// Register REST API route
function register_my_api_routes() {
    register_rest_route('my-plugin/v1', '/posts', [
        'methods' => 'GET',
        'callback' => 'get_custom_posts',
        'permission_callback' => '__return_true'
    ]);
    register_rest_route('my-plugin/v1', '/posts/(?P<id>\d+)', [
        'methods' => 'GET',
        'callback' => 'get_custom_post',
        'args' => [
            'id' => [
                'validate_callback' => function($param) {
                    return is_numeric($param);
                }
            ]
        ]
    ]);
}
add_action('rest_api_init', 'register_my_api_routes');
// API callback functions
function get_custom_posts($request) {
    $posts = get_posts([
        'post_type' => 'portfolio',
        'posts_per_page' => 10
    ]);
    $data = [];
    foreach ($posts as $post) {
        $data[] = [
            'id' => $post->ID,
            'title' => $post->post_title,
            'content' => $post->post_content,
            'featured_image' => get_the_post_thumbnail_url($post->ID)
        ];
    }
    return new WP_REST_Response($data, 200);
}
function get_custom_post($request) {
    $post_id = $request->get_param('id');
    $post = get_post($post_id);
    if (!$post) {
        return new WP_Error('post_not_found', 'Post not found', ['status' => 404]);
    }
    return new WP_REST_Response([
        'id' => $post->ID,
        'title' => $post->post_title,
        'content' => $post->post_content
    ], 200);
}

9.2 Data Validation

Validating User Input
// Validate email
if (!is_email($_POST['email'])) {
    $errors[] = 'Invalid email address';
}
// Validate required fields
if (empty($_POST['name'])) {
    $errors[] = 'Name is required';
}
// Validate numeric input
$age = intval($_POST['age']);
if ($age < 18 || $age > 120) {
    $errors[] = 'Age must be between 18 and 120';
}
// Validate URL
if (!filter_var($_POST['website'], FILTER_VALIDATE_URL)) {
    $errors[] = 'Invalid website URL';
}
// Custom validation function
function validate_username($username) {
    if (strlen($username) < 3) {
        return false;
    }
    if (!preg_match('/^[a-zA-Z0-9_]+$/', $username)) {
        return false;
    }
    return true;
}

9.3 Nonces & CSRF Protection

Using WordPress Nonces
// Create nonce in form
$nonce = wp_create_nonce('my_form_action');
echo '<input type="hidden" name="my_form_nonce" value="' . $nonce . '" />';
// Verify nonce on form submission
if (isset($_POST['my_form_nonce']) &&
    wp_verify_nonce($_POST['my_form_nonce'], 'my_form_action')) {
    // Nonce is valid, process form
    update_post_meta($post_id, 'custom_field', sanitize_text_field($_POST['custom_field']));
} else {
    // Nonce is invalid
    wp_die('Security check failed');
}
// Using nonce in AJAX
function my_ajax_handler() {
    check_ajax_referer('my_ajax_nonce', 'security');
    // Process AJAX request
    $response = ['success' => true, 'message' => 'Data saved'];
    wp_send_json($response);
}
add_action('wp_ajax_my_action', 'my_ajax_handler');

9.4 SQL Injection Prevention

Using $wpdb Safely
global $wpdb;
// SAFE: Using prepare method
$user_id = intval($_GET['user_id']);
$results = $wpdb->get_results(
    $wpdb->prepare(
        "SELECT * FROM {$wpdb->users} WHERE ID = %d",
        $user_id
    )
);
// SAFE: Using placeholders
$post_title = sanitize_text_field($_POST['title']);
$post_content = wp_kses_post($_POST['content']);
$wpdb->insert(
    $wpdb->posts,
    [
        'post_title' => $post_title,
        'post_content' => $post_content,
        'post_status' => 'publish'
    ],
    ['%s', '%s', '%s']
);
// SAFE: Using get_var with prepare
$email = sanitize_email($_POST['email']);
$user_exists = $wpdb->get_var(
    $wpdb->prepare(
        "SELECT COUNT(*) FROM {$wpdb->users} WHERE user_email = %s",
        $email
    )
);

11.4 Migration Scripts

Database Migration Script
// Database migration script
function run_database_migration() {
    global $wpdb;
    // Check if migration already run
    $migration_version = get_option('my_plugin_migration_version', '0');
    if (version_compare($migration_version, '1.1.0', '<')) {
        // Run migration for version 1.1.0
        $sql = "ALTER TABLE {$wpdb->prefix}my_table ADD COLUMN new_field VARCHAR(255) DEFAULT ''";
        $wpdb->query($sql);
        // Update migration version
        update_option('my_plugin_migration_version', '1.1.0');
    }
}
// Search and replace URLs
function update_site_urls($old_url, $new_url) {
    global $wpdb;
    // Update post content
    $wpdb->query(
        $wpdb->prepare(
            "UPDATE {$wpdb->posts} SET post_content = REPLACE(post_content, %s, %s)",
            $old_url,
            $new_url
        )
    );
    // Update post meta
    $wpdb->query(
        $wpdb->prepare(
            "UPDATE {$wpdb->postmeta} SET meta_value = REPLACE(meta_value, %s, %s)",
            $old_url,
            $new_url
        )
    );
    // Update options
    $wpdb->query(
        $wpdb->prepare(
            "UPDATE {$wpdb->options} SET option_value = REPLACE(option_value, %s, %s) WHERE option_name IN ('siteurl', 'home')",
            $old_url,
            $new_url
        )
    );
}
